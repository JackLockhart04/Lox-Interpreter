// Basic arithmetic
(1 + 2) * 3;
1 + 2;
10 - 4;
2 * 3;
9 / 3;

// Grouping and precedence
(1 + 2) * (3 + 4);
(1 + (2 * 3)) - 4;

// Unary
-5;
!false;
!nil;

// Strings and concatenation
"hello" + " " + "world";
"the number four: " + 4;

// Comparisons
3 > 2;
3 >= 3;
2 < 5;
2 <= 1;

// Equality and nil
nil == nil;
nil != 1;
"a" == "a";

// Mixed-type errors (should produce runtime errors but not crash)
2 * (3 / -"muffin");
"a" + 2; // invalid: string + number

// Complex expression to exercise nesting
( (1 + 2) * (3 - 4) ) / ( - ( -2 ) );

// Edge numeric formatting
10.0 + 0.0;
42.5 + 0.5;

// End
